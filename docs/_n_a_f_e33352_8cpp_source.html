53void NAFE33352_Base::DAC::configure( ModeSelect mode, double full_scale_range )
54{
55 uint16_t default_dac_setting[ 6 ] = { 0x0000, 0x1000, 0x87FF, 0x8200, 0xE7FF, 0x0C00 };
56
57 output_mode = mode;
58
59 switch ( output_mode )
60 {
61 case NAFE33352_Base::DAC::ModeSelect::OFF:
62 full_scale = 12.50;
63 default_dac_setting[ 0 ] |= 0x0002;
64 break;
65 case NAFE33352_Base::DAC::ModeSelect::VOLTAGE:
66 full_scale = 12.50;
67 default_dac_setting[ 0 ] |= 0x6040;
68 break;
69 case NAFE33352_Base::DAC::ModeSelect::CURRENT:
70 full_scale = 0.025 / 0.95;
71 default_dac_setting[ 0 ] |= 0x6061;
72 default_dac_setting[ 1 ] |= 0x1000;
73 break;
74 case NAFE33352_Base::DAC::ModeSelect::CURRENT_RECAL:
75 full_scale = 0.025;
76 default_dac_setting[ 0 ] |= 0x6061;
77 default_dac_setting[ 1 ] |= 0x1000;
78 break;
79 case NAFE33352_Base::DAC::ModeSelect::DAC_OFF_VOLTAGE_IN:
80 full_scale = 0.025;
81 default_dac_setting[ 0 ] |= 0x6000;
82 break;
83 case NAFE33352_Base::DAC::ModeSelect::DAC_OFF_CURRENT_IN:
84 full_scale = 0.025;
85 default_dac_setting[ 0 ] |= 0x7000;
86 break;
87 }
88
89 if ( 0.0 < full_scale_range ) // overwrite fullscale range if it is specified
90 full_scale = full_scale_range;
91
92 configure( default_dac_setting );
93}

185void NAFE33352_Base::open_logical_channel( int ch, const uint16_t (&cc)[ 4 ] )
186{
187 static bool pga_enabled = false;
188 constexpr double pow2_24 = (double)(1L << 24);
189 double coeff = 0.00;
190 bool pga_on = false;
191
192 mux_setting[ ch ] = (cc[ 0 ] >> 3) & 0x1F;
193
194 command( NAFE33352_Base::Command::CMD_CH0 + ch );
195
196 switch ( mux_setting[ ch ] )
197 {
198 case VCM_VCM:
199 case GPIO0_GPIO1:
200 coeff = (20.00 * 2.50) / (12.5 * pow2_24);
201 break;
202 case AIP_AIN:
203 coeff = (20.00 * 2.50) / ((cc[ 0 ] & 0x0100 ? 16.00 : 1.00) * pow2_24);
204 pga_on = true;
205 break;
206 case AIP_VSNS:
207 case VSNS:
208 coeff = (20.00 * 2.50) / pow2_24;
209 pga_on = true;
210 break;
211 case AIP:
212 case AIN:
213 coeff = (20.00 * 2.50) / ((cc[ 0 ] & 0x0100 ? 16.00 : 1.00) * pow2_24);
214 pga_on = true;
215 break;
216 case ISNS:
217 coeff = (20.00 * 2.50) / (3.7989 * pow2_24);
218 break;
219 case TIA:
220 coeff = (20.00 * 2.50) / (2.50 * pow2_24);
221 break;
222 case GPIO0_VCM:
223 case VCM_GPIO1:
224 case REF_BYP__VCM:
225 case LDO:
226 case DAC_REF:
227 coeff = (20.00 * 2.50) / (12.5 * pow2_24) + 1.50;
228 break;
229 case VCM__REF_BYP:
230 coeff = (20.00 * 2.50) / (12.5 * pow2_24) - 1.50;
231 break;
232 case BG:
233 //coeff = (20.00 * 2.50) / (12.5 * pow2_24) - 1.50;
234 break;
235 case VADD:
236 coeff = (2.00 * 20.00 * 2.50) / (12.5 * pow2_24) - 1.50;
237 break;
238 case VHDD:
239 case VHSS:
240 coeff = (40.00 * 20.00 * 2.50) / (12.5 * pow2_24);
241 break;
242 }
243
244 coeff_V[ ch ] = coeff;
245
246 if ( pga_on && !pga_enabled )
247 {
248 reg( NAFE33352_Base::Register16::AI_SYSCFG, 0x0800 );
249 pga_enabled = true;
250 }
251
252 for ( auto i = 0; i < 3; i++ )
253 reg( NAFE33352_Base::Register16::AI_CONFIG0 + i, cc[ i ] );
254
255 enable_logical_channel( ch );
256
257 ch_delay[ ch ] = calc_delay( ch );
258}

285double NAFE33352_Base::calc_delay( int ch )
286{
287 constexpr static double data_rates[] = { 288000, 192000, 144000, 96000, 72000, 48000, 36000, 24000,
288 18000, 12000, 9000, 6000, 4500, 3000, 2250, 1125,
289 562.5, 400, 300, 200, 100, 60, 50, 30,
290 25, 20, 15, 10, 7.5, };
291 constexpr static uint16_t delays[] = { 0, 2, 4, 6, 8, 10, 12, 14,
292 16, 18, 20, 28, 38, 40, 42, 56,
293 64, 76, 90, 128, 154, 178, 204, 224,
294 256, 358, 512, 716,
295 1024, 1664, 3276, 7680, 19200, 23040, };
296
297 command( NAFE33352_Base::Command::CMD_CH0 + ch );
298
299 uint16_t ch_config1 = reg( NAFE33352_Base::Register16::AI_CONFIG1 );
300 uint16_t ch_config2 = reg( NAFE33352_Base::Register16::AI_CONFIG2 );
301
302 uint8_t adc_data_rate = (ch_config1 >> 3) & 0x001F;
303 uint8_t adc_sinc = (ch_config1 >> 0) & 0x0007;
304 uint8_t ch_delay = (ch_config2 >> 10) & 0x003F;
305 bool adc_normal_setting = (ch_config2 >> 9) & 0x0001;
306 bool ch_chop = (ch_config2 >> 7) & 0x0001;
307
308 double base_freq = data_rates[ adc_data_rate ];
309 double delay_setting = delays[ ch_delay ] / 4608000.00;
310
311 if ( highspeed_variant )
312 {
313 base_freq *= 2.00;
314 delay_setting /= 2.00;
315 }
316
317 if ( (28 < adc_data_rate) || (4 < adc_sinc) || ((adc_data_rate < 12) && (adc_sinc)) )
318 return 0.00;
319
320 if ( !adc_normal_setting )
321 base_freq /= (adc_sinc + 1);
322
323 if ( ch_chop )
324 base_freq /= 2;
325
326#if 0
327 printf( "adc_data_rate = %d\r\n", adc_data_rate );
328 printf( "base_freq = %lf\r\n", base_freq );
329 printf( "delay_setting = %lf\r\n", delay_setting );
330 printf( "channel delay = %lf\r\n", (1 / base_freq) + delay_setting );
331#endif
332
333 return (1 / base_freq) + delay_setting;
334}

412int32_t NAFE33352_Base::dac_code( double a, double full_scale, uint8_t bit_length )
413{
414 int32_t fsv = (1L << (bit_length - 1));
415
416 int32_t v = (int32_t)((double)fsv * -a / full_scale);
417
418 v = v < -fsv ? -fsv : v;
419 v = v > (fsv - 1) ? (fsv - 1) : v;
420
421 return v << (24 - bit_length);
422}
