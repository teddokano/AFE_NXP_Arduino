253void NAFE13388_Base::open_logical_channel( int ch, const uint16_t (&cc)[ 4 ] )
254{
255 command( ch );
256
257 if ( cc[ 0 ] & 0x0010 )
258 {
259 coeff_uV[ ch ] = ((10.0 / (double)(1L << 24)) / pga_gain[ (cc[ 0 ] >> 5) & 0x7 ]) * 1e6;
260 mux_setting[ ch ] = HV_MUX;
261 }
262 else
263 {
264 coeff_uV[ ch ] = ((10.0 / (double)(1L << 24)) / 2.5) * 1e6;
265 mux_setting[ ch ] = (cc[ 0 ] >> 1) & 0x7;
266 }
267
268 for ( auto i = 0; i < 4; i++ )
269 reg( Register16::CH_CONFIG0 + i, cc[ i ] );
270
271 enable_logical_channel( ch );
272
273 ch_delay[ ch ] = calc_delay( ch );
274}

301double NAFE13388_Base::calc_delay( int ch )
302{
303 constexpr static double data_rates[] = { 288000, 192000, 144000, 96000, 72000, 48000, 36000, 24000,
304 18000, 12000, 9000, 6000, 4500, 3000, 2250, 1125,
305 562.5, 400, 300, 200, 100, 60, 50, 30,
306 25, 20, 15, 10, 7.5, };
307 constexpr static uint16_t delays[] = { 0, 2, 4, 6, 8, 10, 12, 14,
308 16, 18, 20, 28, 38, 40, 42, 56,
309 64, 76, 90, 128, 154, 178, 204, 224,
310 256, 358, 512, 716,
311 1024, 1664, 3276, 7680, 19200, 23040, };
312
313 command( ch );
314
315 uint16_t ch_config1 = reg( Register16::CH_CONFIG1 );
316 uint16_t ch_config2 = reg( Register16::CH_CONFIG2 );
317
318 uint8_t adc_data_rate = (ch_config1 >> 3) & 0x001F;
319 uint8_t adc_sinc = (ch_config1 >> 0) & 0x0007;
320 uint8_t ch_delay = (ch_config2 >> 10) & 0x003F;
321 bool adc_normal_setting = (ch_config2 >> 9) & 0x0001;
322 bool ch_chop = (ch_config2 >> 7) & 0x0001;
323
324 double base_freq = data_rates[ adc_data_rate ];
325 double delay_setting = delays[ ch_delay ] / 4608000.00;
326
327 if ( highspeed_variant )
328 {
329 base_freq *= 2.00;
330 delay_setting /= 2.00;
331 }
332
333 if ( (28 < adc_data_rate) || (4 < adc_sinc) || ((adc_data_rate < 12) && (adc_sinc)) )
334 return 0.00;
335
336 if ( !adc_normal_setting )
337 base_freq /= (adc_sinc + 1);
338
339 if ( ch_chop )
340 base_freq /= 2;
341
342#if 0
343 printf( "base_freq = %lf\r\n", base_freq );
344 printf( "delay_setting = %lf\r\n", delay_setting );
345 printf( "total delay = %lf\r\n", (1 / base_freq) + delay_setting );
346#endif
347
348 return (1 / base_freq) + delay_setting;
349}

469void NAFE13388_Base::gain_offset_coeff( const ref_points &ref )
470{
471 constexpr double pga1x_voltage = 5.0;
472 constexpr int adc_resolution = 24;
473 constexpr double pga_gain_setting = 0.2;
474
475 constexpr double fullscale_voltage = pga1x_voltage / pga_gain_setting;
476
477 double fullscale_data = pow( 2, (adc_resolution - 1) );
478 double ref_data_span = ref.high.data - ref.low.data;
479 double ref_voltage_span = ref.high.voltage - ref.low.voltage;
480
481 double dv_slope = ref_data_span / ref_voltage_span;
482 double custom_gain = dv_slope * (fullscale_voltage / fullscale_data);
483 double custom_offset = (dv_slope * ref.low.voltage - ref.low.data) / custom_gain;
484
485 int32_t gain_coeff_cal = reg( Register24::GAIN_COEFF0 + ref.cal_index );
486 int32_t offsset_coeff_cal = reg( Register24::OFFSET_COEFF0 + ref.cal_index );
487 int32_t gain_coeff_new = round( gain_coeff_cal * custom_gain );
488 int32_t offset_coeff_new = round( custom_offset - offsset_coeff_cal );
489
490#if 0
491 printf( "ref_point_high = %8ld @%6.3lf\r\n", ref.high.data, ref.high.voltage );
492 printf( "ref_point_low = %8ld @%6.3lf\r\n", ref.low.data, ref.low.voltage );
493 printf( "gain_coeff_new = %8ld\r\n", gain_coeff_new );
494 printf( "offset_coeff_new = %8ld\r\n", offset_coeff_new );
495#endif
496
497 reg( Register24::GAIN_COEFF0 + ref.coeff_index, gain_coeff_new );
498 reg( Register24::OFFSET_COEFF0 + ref.coeff_index, offset_coeff_new );
499}

501int NAFE13388_Base::self_calibrate( int pga_gain_index, int channel_selection, int input_select, double reference_source_voltage, bool use_positive_side )
502{
503 constexpr auto low_gain_index = 2;
504 auto channel_in_use = false;
505 ch_setting_t tmp_ch_config;
506 int gain_index = static_cast<int>( pga_gain_index );
507
508 // logical channel selection to perform the self-calibration
509 // if the chennel in-use, save channel setting to temporal memory
510
511 if ( reg( Register16::CH_CONFIG4 ) & (0x1 << channel_selection) )
512 {
513 channel_in_use = true;
514
515 command( channel_selection );
516
517 for ( auto i = 0; i < 4; i++ )
518 tmp_ch_config[ i ] = reg( Register16::CH_CONFIG0 + i );
519 }
520
521 // if user doesn't specify the channel and voltage, use REFH or REFL
522
523 if ( !input_select )
524 {
525 bool low_gain = (gain_index <= low_gain_index);
526
527 input_select = low_gain ? 0x5 : 0x6;
528 reference_source_voltage = (reg( low_gain ? Register24::OPT_COEF1 : Register24::OPT_COEF2 ) * 5.00) / (double)(1UL << 24);
529
530#if 1
531 printf( "==== self-calibration for PGA gain setting: x%3.1lf\r\n", pga_gain[ gain_index ] );
532 printf( "gain = %s\r\n", low_gain ? "low" : "high" );
533 printf( "REF%s = %10.8lfV\r\n", low_gain ? "H" : "L", reference_source_voltage );
534#endif
535 }
536
537 // logical channel settings
538 // Total 3 settings are prepared to measure reference_voltage, internal-GND and AICOM
539
540 const uint16_t REF_GND = 0x0011 | (gain_index << 5);
541 const uint16_t REF_V = (input_select << (use_positive_side ? 12 : 8)) | REF_GND;
542 const uint16_t REF_COM = 0x7700 | REF_GND;
543 const uint16_t ch_config1 = (gain_index << 12) | 0x00E4;
544 constexpr uint16_t ch_config2 = 0x8480;
545 constexpr uint16_t ch_config3 = 0x0000;
546
547 const ch_setting_t refh = { REF_V, ch_config1, ch_config2, ch_config3 };
548 const ch_setting_t refg = { REF_GND, ch_config1, ch_config2, ch_config3 };
549 const ch_setting_t refc = { REF_COM, ch_config1, ch_config2 & ~0x0080, ch_config3 }; //CH_CHOP:off
550
551 // forcing to set unity-gain and zero-offset
552
553 constexpr raw_t default_gain_coeff_value = 0x1UL << 22;
554 constexpr raw_t default_offset_coeff_value = 0;
555
556 reg( Register24::GAIN_COEFF0 + gain_index, default_gain_coeff_value );
557 reg( Register24::OFFSET_COEFF0 + gain_index, default_offset_coeff_value );
558
559 // measure the logical channel with those different 3 settings
560
561 open_logical_channel( channel_selection, refh );
562 raw_t data_REF = start_and_read( channel_selection );
563
564 open_logical_channel( channel_selection, refg );
565 raw_t data_GND = start_and_read( channel_selection );
566
567 open_logical_channel( channel_selection, refc );
568 raw_t data_COM = start_and_read( channel_selection );
569
570 // calculation
571
572 const double fullscale_voltage = 5.00 / pga_gain[ gain_index ];
573 const double calibrated_gain = (double)(0x1UL << 23) * (reference_source_voltage / fullscale_voltage) / (double)(data_REF - data_GND);
574
575#if 0
576 printf( "data_REF = %8ld (%lfV)\r\n", data_REF, raw2v( channel_selection, data_REF ) );
577 printf( "data_GND = %8ld (%lfmV)\r\n", data_GND, raw2mv( channel_selection, data_GND ) );
578 printf( "data_COM = %8ld (%lfmV)\r\n", data_COM, raw2mv( channel_selection, data_COM ) );
579 printf( "gain adjustment = %8lf (%lfdB)\r\n\r\n", calibrated_gain, 20 * log10( calibrated_gain ) );
580#endif
581
582 if ( !( (0.95 < calibrated_gain) && (calibrated_gain < 1.05) ) )
583 return CalibrationError::GainError;
584
585 const double offset_mv = raw2mv( channel_selection, data_COM );
586
587 if ( !( (-10.0 < offset_mv) && (offset_mv < 10.0) ) )
588 return CalibrationError::OffsetError;
589
590 // setting registers: GAIN_COEFF[n] and OFFSET_COEFF[n]
591
592 reg( Register24::GAIN_COEFF0 + gain_index, (uint32_t)(default_gain_coeff_value * calibrated_gain) );
593 reg( Register24::OFFSET_COEFF0 + gain_index, default_offset_coeff_value + data_COM );
594
595 // if the channel was in-use, revert the setting
596
597 if ( channel_in_use )
598 open_logical_channel( channel_selection, tmp_ch_config );
599 else
600 close_logical_channel( channel_selection );
601
602 return CalibrationError::NoError;
603}
