164 inline double raw2uv( int ch, raw_t value )
165 {
166 double v = value * coeff_uV[ ch ];
167
168 if ( HV_MUX != mux_setting[ ch ] )
169 {
170#if 1
171 switch ( mux_setting[ ch ] )
172 {
173 case REF2_REF2:
174 case GPIO0_GPIO1:
175 return v;
176 break;
177 case REFCOARSE_REF2:
178 case VADD_REF2:
179 return 2.00 * (v + 1.5e6);
180 break;
181 case VHDD_REF2:
182 return 32.00 * (v + 0.25e6);
183 break;
184 case REF2_VHSS:
185 return -32.00 * (v - 0.25e6);
186 break;
187 }
188#else
189 return v;
190#endif
191 }

193 return v;
194 }
